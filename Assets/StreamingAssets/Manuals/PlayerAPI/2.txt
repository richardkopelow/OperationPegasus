<size=30><b>File</b></size>

public File(string path);
<i>creates an object for interacting with files.</i>

public string Read();
<i>Returns the contents of the file as a string.</i>

public void Write(string text);
<i>overwrites the contents of the file with text.</i>

public void WriteLine(string text);
<i>Appends text to the contents of the file and makes a new line.</i>

public void Delete();
<i>Deletes the file.</i>
